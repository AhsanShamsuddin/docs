### Dynamic Ads Parameters

<ParamField body="campaignType" type="string" required>
	For Phone Call Goal, campaignType is `LEAD_GENERATION`
</ParamField>

<ParamField body="images" type="object[]">
	<Note>
		You can add up to 10 total images and videos in one dynamic ad, atlease 1 image is required if video is not provided
	</Note>
	<Expandable title="Image Assets Array">
		<ParamField body="url" type="string" required>
			URL for image asset
		</ParamField>
	</Expandable>

</ParamField>

<ParamField body="videos" type="object[]">
	<Note>
		You can add up to 10 total images and videos in one dynamic ad, atlease 1 image is required if video is not provided
	</Note>
	<Expandable title="Video Assets Array">
		<ParamField body="video_id" type="string" required>
			Facebook video ID, get facebook video id from [here](/api-reference/endpoint/meta/get_ad_account_media)
		</ParamField>
		<ParamField body="thumbnail_url" type="string" required>
			URL for video thumbnail image, get video thumbnail url from [here](/api-reference/endpoint/meta/get_ad_account_media)
		</ParamField>
	</Expandable>

</ParamField>

<ParamField body="carousel" type="object[]">
	<Expandable title="Carousel Array">
		<ParamField body="name" type="string" required>
			Carousel Name
		</ParamField>
		<ParamField body="description" type="string">
			Carousel Description
		</ParamField>
		<ParamField body="picture" type="string">
			Carousel Picture URL
		</ParamField>
		<ParamField body="video_id" type="string">
			Carousel Video ID
		</ParamField>
		<ParamField body="link" type="string">
			Carousel Link URL
		</ParamField>
	</Expandable>

</ParamField>

<ParamField body="primaryText" type="object[]">
	<Note>
		You can add up to 5 primary text variations, atleast 1 primary text is required
	</Note>
	<Expandable title="Primary Text Array">
		<ParamField body="text" type="string" required>
			Main ad copy text
		</ParamField>
	</Expandable>

</ParamField>

<ParamField body="headline1" type="object[]">
	<Note>
		You can include up to 5 headline variations.
	</Note>
	<Expandable title="Headline Variations Array">
		<ParamField body="text" type="string" required>
			Headline text variation
		</ParamField>
	</Expandable>

</ParamField>

<ParamField body="phoneNumber" type="string">
	Phone Number
</ParamField>

<ParamField body="greetingText" type="string">
	Greeting Text
</ParamField>

<ParamField body="conversations" type="object[]">
	<Expandable title="Conversations Array">
		<ParamField body="title" type="string" required>
			Conversation Title
		</ParamField>
	</Expandable>
</ParamField>

<ParamField body="isCallAd" type="boolean">
	Whether it's a call ad, if true, `phoneNumber` is required
</ParamField>
